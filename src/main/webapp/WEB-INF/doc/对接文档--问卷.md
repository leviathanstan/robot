  ### 添加问卷（post）
  - 项目地址
  ``````
  <项目地址>survey/addSurvey
  ``````
  - 请求参数(json)
  ```
   {
       "title":"",  //标题
       "remark":"", //备注
       "templateType":""    //问卷：0，模板：1
   }
  ```
  - 返回参数
  ``````
  成功
  "result":"success",
   "message":{
   
             }  
   失败
   "result":"error",
   "message":null                    
  ``````
 ### 添加题目（post）
 - 项目地址
 ``````
 <项目地址>survey/addQuestion
 ``````
 - 请求参数(json)
 ``````
  {
      "surveyId":1,         //问卷id
      "serialNumber":1,     //题目序号                              没有发送0过来
      "title":"first test", //题目标题
      "answerType":0,       //单选：0，多选：1，文字题：2，打分题：3
      "optionType":0,       //必选：0，可选：1
      "minNumber":1,        //最少选项数（多选）、最少字数（文字题）    没有发送0过来
      "maxNumber":3,        //最多选项数（多选）、最少字数（文字题）    没有发送0过来
      "precedentChoiceId":,//关联选项id,                            **没有不要发送过来**
      "choices":            //选项
      [
          {
          "serialNumber":1,         //序号                        没有发送0过来
          "title":"first choice",   //标题
          "type":0,                  //选项型:0、文字型:1
          "defaultStatus":0         //默认状态                      没选就发0
          },
          {
          "serialNumber":2,
          "title":"second choice",
          "type":0，
          "defaultStatus":0         //默认状态                      没选就发0
          },
          {
          "serialNumber":3,
          "title":"third choice",
          "type":0,
          "defaultStatus":0         //默认状态                      没选就发0
          },
          {
          "serialNumber":4,
          "title":"fouth choice",
          "type":1,
          "defaultStatus":0         //默认状态                      没选就发0

          }
      ]
  }
 ``````
 - 返回参数
 ``````
 成功
 "result":"success",
  "message":{
  
            }  
  失败
  "result":"error",
  "message":null                    
 ``````
  ### 回答问题（post）
  - 项目地址
  ``````
  <项目地址>survey/addAnswer
  ``````
  - 请求参数(json)
  ``````
   [
        {
        "questionId":1, //文字题或打分题，单选多选不需要
        "choiceId":1,   //单选或多选时
        "content":"abc" //文字选项或打分题(一分：1)时，单选多选不需要
        },
                {
        "questionId":1,
        "choiceId":1,
        "content":"abc"
        },
                {
        "questionId":1,
        "choiceId":1,
        "content":"abc"
        },
        {
        "questionId":1,
        "choiceId":1,
        "content":"abc"
        }
    ]
  ``````
  - 返回参数
  ``````
  成功
  "result":"success",
   "message":{
   
             }  
   失败
   "result":"error",
   "message":null                    
  ``````
